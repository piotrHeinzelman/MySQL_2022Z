package pl.heinzelman.dbmodule.dict;


import jakarta.persistence.*;
import lombok.Getter;
import lombok.Setter;
import lombok.ToString;
import pl.heinzelman.dbmodule.keyLogic.IdIL;

import java.io.Serializable;

@Getter
@Setter
@ToString

@Entity(name = "icon_name")
@Table(name = "icon_name")
@Inheritance ( strategy = InheritanceType.TABLE_PER_CLASS )
@IdClass( IdIL.class )
public class IconName extends AbsSimpleDict {

    @Column( name = "id_lang")
    private Long ID_icon;

    @Column( name = "id_lang")
    private Long ID_lang;



    public IconName() {}

    public IconName(String name, Long ID_icon, Long ID_lang) {
        super(name);
        this.ID_icon = ID_icon;
        this.ID_lang = ID_lang;
    }
}
